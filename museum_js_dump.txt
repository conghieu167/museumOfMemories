    const controlsToAnimate =
      sectionToObserve.querySelector("#exhibit-controls");

    const sectionObserver = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            // Animate tiÃªu Ä‘á»
            if (headerToAnimate) {
              headerToAnimate.classList.add("is-visible");
            }
            // Má»šI: Animate khu vá»±c nÃºt báº¥m

      {

      }

    timelineScrollContainer.addEventListener('wheel', (e) => {

      bokeh.style.height = size + "px";
      bokeh.style.left = (Math.random() * 100) + "%";
      bokeh.style.top = (Math.random() * 100) + "%";
      bokeh.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
      bokeh.style.animationDuration = (Math.random() * 10 + 15) + "s";
      bokeh.style.animationDelay = (Math.random() * 5) + "s";
      timelineBg.appendChild(bokeh);
    }

    // 2. Tạo Đom đóm / Bụi sao bay lơ lửng
    for (let i = 0; i < 40; i++) {
      const firefly = document.createElement("div");
      firefly.className = "firefly";
      const size = Math.random() * 3 + 1; // 1 - 4px
      firefly.style.width = size + "px";
      firefly.style.height = size + "px";
      firefly.style.left = (Math.random() * 120) + "%"; // Khởi tạo rộng ra
      firefly.style.top = (Math.random() * 120) + "%";
      
      const duration = Math.random() * 15 + 10;
      firefly.style.animationDuration = duration + "s";
      firefly.style.animationDelay = (Math.random() * 10) + "s";
      
      const endX = -Math.random() * 300 - 100; // Bay sang trái
      const endY = (Math.random() - 0.5) * 400; // Bay lên/xuống nhẹ
      firefly.style.setProperty("--end-x", endX + "px");
      firefly.style.setProperty("--end-y", endY + "px");
      
      timelineBg.appendChild(firefly);
    }
  }
